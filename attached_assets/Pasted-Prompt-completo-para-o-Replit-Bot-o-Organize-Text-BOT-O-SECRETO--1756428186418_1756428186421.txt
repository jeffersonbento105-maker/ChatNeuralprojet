Prompt completo para o Replit – Botão "Organize Text"

/* =========================
  BOTÃO SECRETO ORGANIZE TEXT
========================= */

// 1. Criação do botão discreto
const organizeButton = document.createElement("button");
organizeButton.id = "organizeTextBtn";
organizeButton.innerText = "Organize Text"; // Nome em inglês
organizeButton.style.position = "fixed";
organizeButton.style.bottom = "10px"; // Ajuste a posição discreta
organizeButton.style.right = "10px";
organizeButton.style.padding = "4px 8px";
organizeButton.style.fontSize = "10px";
organizeButton.style.backgroundColor = "#eee";
organizeButton.style.color = "#333";
organizeButton.style.border = "1px solid #aaa";
organizeButton.style.borderRadius = "4px";
organizeButton.style.cursor = "pointer";
organizeButton.style.zIndex = "9999";
organizeButton.style.opacity = "0.5"; // Discreto
organizeButton.title = "Click to organize the AI text"; // Tooltip em inglês

// 2. Adiciona o botão à tela
document.body.appendChild(organizeButton);

// 3. Função que reorganiza o texto
function organizeText() {
  // Supondo que o texto da IA esteja dentro de um div com id="aiOutput"
  const aiOutput = document.getElementById("aiOutput");
  if (!aiOutput) return;

  let text = aiOutput.innerText;

  // Remove quebras de linha extras
  text = text.replace(/\n{2,}/g, "\n");

  // Adiciona numeração simples por linha
  const lines = text.split("\n").map((line, index) => {
    if (line.trim() === "") return "";
    return ${index + 1}. ${line.trim()};
  });

  // Atualiza o texto organizado
  aiOutput.innerHTML = lines.join("<br>");
}

// 4. Conecta o botão à função
organizeButton.addEventListener("click", organizeText);


---

✅ Como funciona:

1. Cria um botão pequeno e discreto no canto da tela (bottom:10px, right:10px), quase secreto.


2. Ao clicar, pega o texto que está no div com id="aiOutput" (você pode alterar para o id do seu output real).


3. Remove linhas duplicadas e espaços extras.


4. Numera cada linha automaticamente.


5. Atualiza o texto na tela, mantendo cores, estilo e legibilidade.
